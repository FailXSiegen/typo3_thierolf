function centerActiveBulletWithTransform(glideElement) { if(window.innerWidth <= 768) { var $activeBullet = $(glideElement).find('.glide__bullet--active'); // Selektieren des aktiven Bullets var $bulletsContainer = $(glideElement).find('.glide__bullets'); // Bullet-Container var activeBulletOffset = $activeBullet.position().left + $activeBullet.outerWidth() / 2; // Zentrum des aktiven Bullets var containerCenter = $bulletsContainer.width() / 2; // Zentrum des Containers var transformValue = containerCenter - activeBulletOffset; // Berechnung des Transform-Werts // Anwendung der Transformation auf den Container $bulletsContainer.css('transform', 'translateX(' + transformValue + 'px)'); } } var glideElement = '#gslider{f:if(condition:data._LOCALIZED_UID, then:data._LOCALIZED_UID, else:data.uid)}'; var glide = new Glide(glideElement, { type: 'carousel', autoplay: false, perView: 1, focusAt: 'center', peek: { before: 100, after: 100 } }).mount() glide.on('move.after', function() { centerActiveBulletWithTransform(glideElement); // Aufruf der Funktion zum Zentrieren des aktiven Bullets }); // Initialer Aufruf f√ºr den Fall, dass das aktive Bullet beim Laden der Seite zentriert werden muss centerActiveBulletWithTransform(glideElement);
